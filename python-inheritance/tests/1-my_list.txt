# Test cases for MyList class

# Import the class MyList
>>> MyList = __import__('1-my_list').MyList

Test 1: Empty list
>>> my_list = MyList()
>>> my_list.print_sorted()
[]

Test 2: List with positive integers
>>> my_list = MyList([5, 2, 8, 1, 9])
>>> my_list.print_sorted()
[1, 2, 5, 8, 9]

Test 3: List with negative and positive integers
>>> my_list = MyList([-4, 3, 0, -1, 2])
>>> my_list.print_sorted()
[-4, -1, 0, 2, 3]

Test 4: List with already sorted elements
>>> my_list = MyList([1, 2, 3, 4, 5])
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

Test 5: List with elements in descending order
>>> my_list = MyList([5, 4, 3, 2, 1])
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
